# ---------- File Handling ----------

def load_users():
    users = {}
    if os.path.exists(USERS_FILE):
        with open(USERS_FILE, "r") as f:
            for line in f:
                if "," in line:
                    u, p, v = line.strip().split(",")
                    users[u] = {"password": p, "voted": v}
    return users

def save_users(users):
    with open(USERS_FILE, "w") as f:
        for u, d in users.items():
            f.write(f"{u},{d['password']},{d['voted']}\n")

def load_votes():
    votes = {c: 0 for c in candidates}
    if os.path.exists(VOTES_FILE):
        with open(VOTES_FILE, "r") as f:
            for line in f:
                if "," in line:
                    name, count = line.strip().split(",")
                    votes[name] = int(count)
    return votes

def save_votes(votes):
    with open(VOTES_FILE, "w") as f:
        for name, count in votes.items():
            f.write(f"{name},{count}\n")

                btn.pack(pady=10)

    def cast_vote(self, choice):
        votes = load_votes()
        users = load_users()
        u = self.controller.current_user

        votes[choice] += 1
        users[u]["voted"] = "yes"
        save_votes(votes)
        save_users(users)

        messagebox.showinfo("Success", f"Vote for {choice} Casted Successfully!")
        self.on_show()

        names = list(votes.keys())
        counts = list(votes.values())
        
        if sum(counts) > 0:
            max_votes = max(counts)
            winners = [n for n, c in votes.items() if c == max_votes]
            self.winner_label.configure(text=f"👑 Current Leader: {', '.join(winners)} ({max_votes} votes)")

            ax.set_ylabel('Total Votes', fontsize=14)

            txt = "\n".join([f"{k}: {v} votes" for k, v in votes.items()])

    def export_csv(self):
        votes = load_votes()
        filename = f"election_results_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"

            writer.writerow(['Candidate', 'Votes'])
            for n, c in votes.items():
                writer.writerow([n, c])
        messagebox.showinfo("Success", f"Results exported to {filename}")

            widget.destroy()

        if not u: return
        
        users = load_users()
        if users[u]["voted"] == "yes":
            ctk.CTkLabel(self.buttons_frame, text="You have already cast your vote!", font=ctk.CTkFont(size=20, weight="bold"), text_color="red").pack(pady=30)
        else:
            for c in candidates:

        votes = load_votes()
        users = load_users()
        u = self.controller.current_user

        votes[choice] += 1
        users[u]["voted"] = "yes"
        save_votes(votes)
        save_users(users)

        messagebox.showinfo("Success", "Vote Casted Successfully!")
        self.on_show()

    def __init__(self, parent, controller):
        super().__init__(parent, fg_color="transparent")
        self.controller = controller
        
        self.header = ctk.CTkFrame(self, fg_color="transparent")

        btn_frame.pack(side="right")
        
        ctk.CTkButton(btn_frame, text="Export CSV", font=ctk.CTkFont(size=14, weight="bold"), fg_color="#4CAF50", hover_color="#45a049", width=120, command=self.export_csv).pack(side="left", padx=5)
        ctk.CTkButton(btn_frame, text="Reset Election", font=ctk.CTkFont(size=14, weight="bold"), fg_color="#E91E63", hover_color="#C2185B", width=120, command=self.reset_election).pack(side="left", padx=5)
        
        self.winner_label = ctk.CTkLabel(self, text="Current Leader: TBD", font=ctk.CTkFont(size=20, weight="bold"), text_color="orange")
        self.winner_label.pack(pady=5)
        
        self.graph_frame = ctk.CTkFrame(self, fg_color="transparent")
        self.graph_frame.pack(fill="both", expand=True, padx=20, pady=10)

    def on_show(self):
        self.draw_graph()

    def draw_graph(self):
        for w in self.graph_frame.winfo_children():
            w.destroy()
            
        votes = load_votes()
        names = list(votes.keys())
        counts = list(votes.values())
        
        if sum(counts) > 0:
            max_votes = max(counts)
            winners = [n for n, c in votes.items() if c == max_votes]
            self.winner_label.configure(text=f"Current Leader: {', '.join(winners)} ({max_votes} votes)")
        else:
            self.winner_label.configure(text="Current Leader: No votes cast yet")

        if MATPLOTLIB_AVAILABLE:
            fig = Figure(figsize=(6, 4), dpi=100)
            fig.patch.set_facecolor('#2b2b2b')
            ax = fig.add_subplot(111)
            ax.set_facecolor('#2b2b2b')
            ax.tick_params(colors='white')
            ax.xaxis.label.set_color('white')
            ax.yaxis.label.set_color('white')
            
            ax.bar(names, counts, color=['#4CAF50', '#FF9800', '#2196F3', '#E91E63'])
            ax.set_ylabel('Votes')
            
            canvas = FigureCanvasTkAgg(fig, master=self.graph_frame)
            canvas.draw()
            canvas.get_tk_widget().pack(fill=ctk.BOTH, expand=True)
        else:
            ctk.CTkLabel(self.graph_frame, text="Matplotlib not installed.", text_color="orange").pack()
            txt = "\n".join([f"{k}: {v} votes" for k, v in votes.items()])
            ctk.CTkLabel(self.graph_frame, text=txt, font=ctk.CTkFont(size=18)).pack(pady=20)

    def export_csv(self):
        votes = load_votes()
        filename = f"election_results_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"
        with open(filename, mode='w', newline='') as file:
            writer = csv.writer(file)
            writer.writerow(['Candidate', 'Votes'])
            for n, c in votes.items():
                writer.writerow([n, c])
        messagebox.showinfo("Success", f"Results exported to {filename}")
        
    def reset_election(self):
        if messagebox.askyesno("Warning", "Are you sure you want to wipe all votes and user statuses? This cannot be undone."):
            empty_votes = {c: 0 for c in candidates}
            save_votes(empty_votes)
            
            users = load_users()
            for u in users:
                users[u]["voted"] = "no"
            save_users(users)
            
            messagebox.showinfo("Success", "Election data has been completely reset.")
            self.on_show()
